% ── Custom Commands ───────────────────────────────────────────────────────────
% Two-column entry: {Company/School}{Date}
\newcommand{\resumeEntry}[4]{
  \vspace{-2pt}\item
    \begin{tabular*}{0.97\textwidth}[t]{l@{\extracolsep{\fill}}r}
      \textbf{#1} & #2 \\
      \textit{\small#3} & \textit{\small #4} \\
    \end{tabular*}\vspace{-7pt}
}

  \vspace{-2pt}\item

}
